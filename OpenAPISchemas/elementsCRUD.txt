openapi: 3.0.0
info:
  title: Datasource CRUD API
  description: Azure Function for managing various datasource elements via CRUD operations
  version: 1.0.0
servers:
  - url: /api/{element}
paths:
  /:
    get:
      summary: Retrieve an element or list all elements
      parameters:
        - name: element
          in: path
          required: true
          description: The target datasource element
          schema:
            type: string
        - name: id
          in: query
          required: false
          description: ID of the element to retrieve. If not specified, all elements are returned.
          schema:
            type: string
      responses:
        '200':
          description: Successful response
          content:
            application/json:
              schema:
                type: array
                items:
                  $ref: '#/components/schemas/Element'
        '400':
          description: Invalid request
        '405':
          description: Method not allowed
    post:
      summary: Create or update an element
      parameters:
        - name: element
          in: path
          required: true
          description: The target datasource element
          schema:
            type: string
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/Element'
      responses:
        '200':
          description: Element created or updated successfully
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ElementResponse'
        '400':
          description: Invalid JSON payload
    delete:
      summary: Delete an element
      parameters:
        - name: element
          in: path
          required: true
          description: The target datasource element
          schema:
            type: string
        - name: id
          in: query
          required: true
          description: ID of the element to delete
          schema:
            type: string
      responses:
        '200':
          description: Element deleted successfully
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/DeleteResponse'
        '400':
          description: Invalid JSON payload
        '405':
          description: Method not allowed
components:
  schemas:
    Element:
      type: object
      properties:
        id:
          type: string
          description: The ID of the element
        name:
          type: string
          description: The name of the element
        properties:
          type: array
          items:
            type: string
          description: List of properties associated with the element
    ElementResponse:
      type: object
      properties:
        success:
          type: boolean
        element:
          $ref: '#/components/schemas/Element'
    DeleteResponse:
      type: object
      properties:
        success:
          type: boolean
        message:
          type: string
